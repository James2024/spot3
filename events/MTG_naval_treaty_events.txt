		}

#hidden event to remove treaty if two majors are at war
country_event = {
	id = MTG_naval_treaty.10
	
	is_triggered_only = yes
	
	hidden = yes

	fire_only_once = yes
	
	immediate = {
		every_country = {
			limit = {
				has_naval_treaty_trigger = yes
				has_war = no
			}
			country_event = MTG_naval_treaty.11
		}
		every_country = {
			limit = {
				has_naval_treaty_trigger = yes
			}
			remove_naval_treaty_effect = yes
		}
	}
	option = { #great
	}
}
